  species_sub <- species_hel[idx, ]
  # compute distance for this subset
  dist_sub <- dist(species_sub, method = "euclidean")
  # run PERMANOVA
  adonis2(
    dist_sub ~ Landscape_style,
    data = env[idx, ],
    permutations = 999
  )
})

# 3.Computing distances between species based on functional traits 

# sp_tr = if there are a lot of species with identical traits, compute sp_dist between FEs (fe_tr), not single species (Beetles_traits)

sp_dist <- mFD::funct.dist( 
  sp_tr         = Beetles_traits,
  tr_cat        = traits_metadata, 
  metric        = "gower", # if all traits are continuous use "euclidean" 
  scale_euclid  = "scale_center", # center transformation
  ordinal_var   = "classic", # “classic” preserves the quantitative meaning of the classes

# Function to make pie chart for any trait

make_pie <- function(df, trait_col, title, palette="Set2") {
  df_sum <- df %>%
    group_by(.data[[trait_col]]) %>%
    summarise(n_taxa = n_distinct(Species)) %>%
    ungroup() %>%
    mutate(Proportion = n_taxa / sum(n_taxa),
           label = paste0(round(Proportion*100,1), "%")) %>%
    arrange(desc(.data[[trait_col]])) %>%
    mutate(ymax = cumsum(Proportion),
           ymin = ymax - Proportion,
           label_pos = (ymax + ymin)/2)
  
  ggplot(df_sum, aes(ymax = ymax, ymin = ymin, xmax = 4, xmin = 3, fill = .data[[trait_col]])) +
    geom_rect() +
    coord_polar(theta = "y") +
    geom_text(aes(y = label_pos, x = 3.5, label = label), color = "white", size = 4) +
    labs(title = title, fill = trait_col) +
    scale_fill_brewer(palette = palette) +
    theme_void() +
    theme(legend.position = "right",
          plot.title = element_text(face = "bold", hjust = 0.5))
}

# Function to summarize traits by landscape
trait_summary_landscape <- function(df, trait_col) {
  
  df %>%
    group_by(
      Landscape_style,
      Trait = .data[[trait_col]],
      Origin
    ) %>%
    summarise(
      Taxa = n_distinct(Species),
      .groups = "drop"
    ) %>%
    
    # % within landscape (sums to 100 per landscape)
    group_by(Landscape_style) %>%
    mutate(
      `% within landscape` = round(Taxa / sum(Taxa) * 100, 1)
    ) %>%
    ungroup() %>%
    
    mutate(Trait_type = trait_col) %>%
    arrange(Trait_type, Landscape_style, Trait, Origin) %>%
    select(
      Landscape_style,
      Trait_type,
      Trait,
      Origin,
      Taxa,
      `% within landscape`
    )
}
